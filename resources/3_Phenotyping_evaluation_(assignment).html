Define function to calculate performance.

## getStats(df, predicted, reference)
getStats <- function(df, ...){
  df %>%
    select_(.dots = lazyeval::lazy_dots(...)) %>%
    mutate_all(funs(factor(., levels = c(1,0)))) %>% 
    table() %>% 
    confusionMatrix()
}

Gather data sources

Data source 1: Clinical definition

  • Systolic BP >= 140 mmHg on more than two occasions
  • Diastolic BP>= 90 mmHg on more than two occasions
d_items <- tbl(con, "mimic3_demo.D_ITEMS")
chartevents <- tbl(con, "mimic3_demo.CHARTEVENTS")

systolic_BP_items_id <- d_items %>%
  filter(lower(LABEL) %like% "%systolic%") %>%
  distinct(ITEMID)
diastolic_BP_items_id <- d_items %>%
  filter(lower(LABEL) %like% "%diastolic%") %>%
  distinct(ITEMID)

systolic_BP_above140_twice <- patient_data %>%
  inner_join(chartevents, by = c("SUBJECT_ID" = "SUBJECT_ID"), suffix = c("","_event")) %>% 
  inner_join(systolic_BP_items_id, by = c("ITEMID" = "ITEMID"), suffix = c("_event","_item")) %>% 
  mutate(systolic_BP_above140 = case_when(VALUENUM >= 140 ~ 1, TRUE ~ 0)) %>% 
  group_by(SUBJECT_ID) %>% 
  summarise(n_systolic_events = sum(systolic_BP_above140, na.rm = TRUE)) %>%
  select(SUBJECT_ID, n_systolic_events)

diastolic_BP_above90_twice <- patient_data %>%
  inner_join(chartevents, by = c("SUBJECT_ID" = "SUBJECT_ID"), suffix = c("","_event")) %>% 
  inner_join(diastolic_BP_items_id, by = c("ITEMID" = "ITEMID"), suffix = c("_event","_item")) %>% 
  mutate(diastolic_BP_above90 = case_when(VALUENUM >= 90 ~ 1, TRUE ~ 0)) %>% 
  group_by(SUBJECT_ID) %>% 
  summarise(n_diastolic_events = sum(diastolic_BP_above90, na.rm = TRUE)) %>% 
  select(SUBJECT_ID, n_diastolic_events)

patients_by_clinical_criteria <- systolic_BP_above140_twice %>% 
  inner_join(diastolic_BP_above90_twice, by = c("SUBJECT_ID" = "SUBJECT_ID"), suffix = c("","_dia")) %>%
  mutate(clinical_criterion = case_when(n_systolic_events > 2 && n_diastolic_events > 2 ~ 1, TRUE ~ 0)) %>% 
  select(SUBJECT_ID, clinical_criterion)

Data source 3: Medications

prescriptions <- tbl(con, "mimic3_demo.PRESCRIPTIONS")
antihypertensives <- tbl(con, "course3_data.D_ANTIHYPERTENSIVES") %>%
  pull(DRUG)

patients_by_medications <- patient_data %>%
  inner_join(prescriptions, by = c("SUBJECT_ID" = "SUBJECT_ID"), suffix = c("","_prescriptions")) %>% 
  mutate(took_drug = case_when(DRUG %in% !!antihypertensives ~ 1, TRUE ~ 0)) %>%
  group_by(SUBJECT_ID) %>% 
  summarise(n_medications = sum(took_drug, na.rm = TRUE)) %>% 
  mutate(took_drug = case_when(n_medications >= 1 ~ 1, TRUE ~ 0)) %>%
  select(SUBJECT_ID, took_drug)
